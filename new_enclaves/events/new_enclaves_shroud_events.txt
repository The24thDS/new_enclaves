# neshr.110: What can your guidance offer us?
country_event = {
	id = neshr.110
	title = neshr.110.name
	desc = neshr.110.desc
	location = event_target:ne_shroud_system

	is_triggered_only = yes
	diplomatic = yes
	custom_gui = "enclave_shroud_window"

	picture_event_data = {
		portrait = event_target:ne_shroud
		room = enclave_shroud_room
	}

	immediate = {
		set_country_flag = ne_shroud_diplomacy_engaged
		export_trigger_value_to_variable = { trigger = num_pops variable = shr_guide_price }
		multiply_variable = { which = shr_guide_price value = @SHR_GUIDE_PRICE_PER_POP }
		change_variable = { which = shr_guide_price value = @SHR_GUIDE_MIN_PRICE }
		ceiling_variable = shr_guide_price
	}

	after = {
		hidden_effect = {
			remove_country_flag = ne_shroud_diplomacy_engaged
			if = {
				limit = { is_variable_set = shr_guide_price }
				clear_variable = shr_guide_price
			}
		}
	}

	option = {
		name = neshr.110.a
		response_text = neshr.110.a.reply
		custom_gui = "enclave_shroud_option"
		allow = {
			resource_stockpile_compare = { resource = energy value >= shr_guide_price }
			hidden_trigger = {
				NOT = { has_country_flag = attacked_ne_shroud }
			}
		}
		add_resource = { energy = -1 mult = shr_guide_price }
		custom_tooltip = opinion20
		add_modifier = {
			modifier = shrouds_children_guidance
			days = @SHR_GUIDE_DURATION_DAYS
		}
		hidden_effect = {
			event_target:ne_shroud = {
				add_trust = {
					who = root
					amount = 20
				}
			}
			set_relation_flag = {
				who = event_target:ne_shroud
				flag = ne_shroud_guidance_hired
			}
			set_variable = {
				which = ne_shr_guide_duration
				value = @DEFAULT_SHR_GUIDE_DURATION_YEARS
			}
			country_event = { id = neshr.111 days = @SHR_GUIDE_DURATION_DAYS }
		}
	}

	# Back to Main Menu
	option = {
		name = ne_back
		custom_gui = "enclave_shroud_option"
		hidden_effect = {
			country_event = { id = neshr.101 }
		}
	}

	# End
	option = {
		name = GOODBYE
		custom_gui = "enclave_shroud_option"
		default_hide_option = yes
	}
}

# neshr.111: Continue guiding?
country_event = {
	id = neshr.111
	title = neshr.111.name
	desc = neshr.111.desc
	location = event_target:ne_shroud_system

	is_triggered_only = yes
	diplomatic = yes
	custom_gui = "enclave_shroud_window"

	picture_event_data = {
		portrait = event_target:ne_shroud
		room = enclave_shroud_room
	}

	trigger = {
		NOT = { has_country_flag = attacked_ne_shroud }
		exists = event_target:ne_shroud
	}

	immediate = {
		remove_relation_flag = {
			who = event_target:ne_shroud
			flag = ne_shroud_trade_deal
		}
		# Normal duration cost calculation
		export_trigger_value_to_variable = { trigger = num_pops variable = shr_guide_price }
		multiply_variable = { which = shr_guide_price value = @SHR_GUIDE_PRICE_PER_POP }
		change_variable = { which = shr_guide_price value = @SHR_GUIDE_MIN_PRICE }
		ceiling_variable = shr_guide_price
		# Extended duration calculation
		set_variable = { which = ne_shr_guide_extended_duration value = this.ne_shr_guide_duration }
		change_variable = { which = ne_shr_guide_extended_duration value = 5 }
		# Extended duration cost calculation
		set_variable = { which = shr_extended_guide_price value = this.shr_guide_price }
		multiply_variable = { which = shr_extended_guide_price value = this.ne_shr_guide_extended_duration }
		divide_variable = { which = shr_extended_guide_price value = @DEFAULT_SHR_GUIDE_DURATION_YEARS }
		multiply_variable = { which = shr_extended_guide_price value = 1.2 } # +20%
		# Extended duration opinion calculation
		set_variable = { which = shr_extended_guide_opinion value = this.ne_shr_guide_extended_duration }
		multiply_variable = { which = shr_extended_guide_opinion value = 2 }
	}

	# For 10 years
	option = {
		name = neshr.111.a
		response_text = neshr.111.a.reply
		custom_gui = "enclave_shroud_option"
		allow = {
			resource_stockpile_compare = { resource = energy value >= shr_guide_price }
			hidden_trigger = {
				NOT = { has_country_flag = attacked_ne_shroud }
			}
		}
		add_resource = { energy = -1 mult = shr_guide_price }
		custom_tooltip = opinion20
		add_modifier = {
			modifier = shrouds_children_guidance
			days = @SHR_GUIDE_DURATION_DAYS
		}
		hidden_effect = {
			event_target:ne_shroud = {
				add_trust = {
					who = Root
					amount = 20
				}
			}
			set_relation_flag = {
				who = event_target:ne_shroud
				flag = ne_shroud_guidance_hired
			}
			set_variable = {
				which = ne_shr_guide_duration
				value = @DEFAULT_SHR_GUIDE_DURATION_YEARS
			}
			country_event = { id = neshr.111 days = @SHR_GUIDE_DURATION_DAYS }
		}
	}

	# Extended contract
	option = {
		name = neshr.111.b
		response_text = neshr.111.a.reply
		custom_gui = "enclave_shroud_option"
		allow = {
			resource_stockpile_compare = { resource = energy value >= shr_extended_guide_price }
			hidden_trigger = {
				NOT = { has_country_flag = attacked_ne_shroud }
			}
		}
		add_resource = { energy = -1 mult = shr_extended_guide_price }
		custom_tooltip = neshr.111.variable_opinion
		add_modifier = {
			modifier = shrouds_children_guidance
			days = 360
			time_multiplier = ne_shr_guide_extended_duration
		}
		hidden_effect = {
			event_target:ne_shroud = {
				add_trust = {
					who = root
					amount = prev.shr_extended_guide_opinion
				}
			}
			set_relation_flag = {
				who = event_target:ne_shroud
				flag = ne_shroud_guidance_hired
			}
			set_variable = {
				which = ne_shr_guide_duration
				value = this.ne_shr_guide_extended_duration
			}
			country_event = { id = neshr.112 days = 1801 }
		}
	}

	# End
	option = {
		name = neshr.111.end
		custom_gui = "enclave_shroud_option"
		default_hide_option = yes
	}
}
